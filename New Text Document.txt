local player = game.Players:FindFirstChild("PlayerName") -- replace with actual player name or variable
if player and player.Character then
    local head = player.Character:FindFirstChild("Head")
    if head then
        head.Transparency = 1
        head.CanCollide = false
        local face = head:FindFirstChild("face")
        if face then
            face.Transparency = 1
        end
    end
end
