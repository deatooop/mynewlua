local player = game.Players.LocalPlayer
local character = player.Character or player.CharacterAdded:Wait()
local head = character:WaitForChild("Head")

-- Create attachments for the beam
local attachment0 = Instance.new("Attachment", head)
attachment0.Position = Vector3.new(0, 0, 0) -- at the head

local attachment1 = Instance.new("Attachment")
attachment1.Position = Vector3.new(0, 100, 0) -- 100 studs above the head
attachment1.Parent = workspace.Terrain

-- Create the red beam
local beam = Instance.new("Beam")
beam.Color = ColorSequence.new(Color3.new(1, 0, 0)) -- red color
beam.Attachment0 = attachment0
beam.Attachment1 = attachment1
beam.Width0 = 0.2
beam.Width1 = 0.2
beam.Parent = head

-- Move the head 100 studs up, make it non-collidable and anchored
head.CFrame = head.CFrame + Vector3.new(0, 100, 0)
head.Anchored = true
head.CanCollide = false

-- Optional: keep the head visible (Transparency = 0)
head.Transparency = 0

-- Clean up when character respawns
player.CharacterRemoving:Connect(function()
    attachment0:Destroy()
    attachment1:Destroy()
    beam:Destroy()
end)
